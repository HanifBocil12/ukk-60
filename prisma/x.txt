model users{
  id int @default(autoincrement())
  nama string
  email string
  password string
  image_profile string
  role Role
  
  siswa Siswa?
  guru Guru?
  admin Admin?

  transaksi Transaksi[]
  boking Boking[]

  resent Aduan[] @relation("admin_pengirim")
  recv Aduan[] @relation("walas penerima")
}

model siswa{
  id
  nisn string

  user users @relation()
  kelas clases @relation()
}

model guru{
  id
  nip string

  users users @relation()

  walas walas?
}

model admin{
  id
  nip string

  userid int
  users users @relation()
}

model walas{
  id 
  guruid Int
  guru guru @relation()

  kelas clases[]
}

model clases {
  id
  tinggkat tinggkat
  nama_kelas nama_kelas

  walasid int
  walas walas @relation()

  siswa siswa[]
}

// transaksi--
model Transaksi{
  id 
  tipe Tipe
  tanggal_terima DateTime
  tanggal_kembali DateTime
  status_transaksi status

  userid
  user user @relation()

  detailtransaksi detailtransaksi[]
  boking boking?
  denda denda?
}

model detailtransaksi{
  id
  transaksi Transaksi @relation()
  buku buuku @relation()

  kondisi_buku String
  jumlah Int
}

model Aduan{
  id

  resent users? @relation("admin_pengirim", )
  recv users? @relation("walas_penerima", )

  desc string
  status status_aduan
}

model denda{
  id
  transaksiid
  transaksi transaksi @relation()

  alasan string
  nilaidenda int
}
